Stores Customer Information
CREATE TABLE cust (
    ID INT PRIMARY KEY,
    FIRST_NAME VARCHAR(50) NOT NULL,
    LAST_NAME VARCHAR(50) NOT NULL,
    COUNTRY VARCHAR(50) NOT NULL,
    PH_NO BIGINT NOT NULL UNIQUE,
    ID_PROOF VARCHAR(50) NOT NULL,
    ROOM ENUM('Maharaja', 'Deluxe', 'Non-Deluxe') NOT NULL,
    ROOM_NO INT UNIQUE NOT NULL,
    CHECK_IN DATE NOT NULL
);

Stores Guest Billing Information
CREATE TABLE info (
    ID INT PRIMARY KEY,
    FIRST_NAME VARCHAR(50) NOT NULL,
    SWIMMING INT DEFAULT 0,
    ROOM_SERVICE INT DEFAULT 0,
    BILL INT DEFAULT 0,
    CHECK_OUT DATE DEFAULT NULL,
    FOREIGN KEY (ID) REFERENCES cust(ID) ON DELETE CASCADE
);

Daily Revenue Report
CREATE TABLE revenue (
    DATE DATE PRIMARY KEY,
    TOTAL_PROFIT INT DEFAULT 0,
    ROOM_SERVICE_REVENUE INT DEFAULT 0,
    ROOM_REVENUE INT DEFAULT 0
);
